
-- SELECIONA VÁRIOS DADOS E DÁ NOMES PROVISÓRIOS ÀS TABELAS --
SELECT CD_LOJ "Código da loja", CD_VEN "Venda", CD_PAR "Parcela", 
	   NVL(VL_PAGO,0)-VL_PAR "Diferença"
	   FROM PARCELA;

	   
-- CONCATENA UMA STRING À UM DADOS SELECIONADO DO BANCO --
SELECT NM_CLI || ' reside em ' || NM_CID "Localização"
	   FROM CLIENTE;


-- SELECIONA DETERMINADO DADO DO BANCO SEM REPETÍ-LO --
SELECT DISTINCT NM_CID FROM CLIENTE;


-- SELECIONA DADOS DE ACORDO COM O OPERADOR  LIKE e '_' --
-- Variação NOT LIKE 
SELECT * FROM CLIENTE
WHERE UPPER(NM_CID) LIKE '__O%';


-- SELECIONA DATAS DENTRE UM DETERMINADO RAGE --
-- Variação NOT BETWEEN 
SELECT * FROM VENDA
WHERE DT_VEN NOT BETWEEN '01/01/2006' AND '31/12/2006';


-- SELECIONA SOMENTE OS VENDEDORES QUE  TENHAS OS CÓDIGOS ESPECIFICADOS --
-- Variação NOT IN 
SELECT NM_VDD FROM VENDEDOR
WHERE CD_VDD IN (1, 3, 5, 7);


-- SELECIONA OS VENDEDORES QUE NÃO POSSUEM SUPERVISORES, OU SEJA CD_SUPERVISOR TEM QUE SER NULL --
-- Variação IS NOT NULL 
SELECT NM_VDD FROM VENDEDOR
WHERE CD_SUPERVISOR IS NULL;


-- ORDENA AS CIDADES POR ORDEM CRESCENTE E PARA CADA UMA, ORDENA OS CLIENTES POR ORDEM DECRESCENTE --
SELECT NM_CLI, NM_CID FROM CLIENTE
ORDER BY NM_CID, NM_CLI DESC;
